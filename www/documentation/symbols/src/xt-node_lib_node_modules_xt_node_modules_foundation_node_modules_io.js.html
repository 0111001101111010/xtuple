  4 /** @namespace
  5   All functionality for input/output to logs or console reporting
  6   is included in this namespace. Filesystem manipulation is also
  7   pulled into this namespace for consistency but can also be used
  8   directly.
  9   
 10   This is only partially implemented but abstracted for extension.
 11 */
 12 
 13 xt.io = { /** @lends xt.io */
 14   
 15   /**
 16     Reference to stdout.
 17     
 18     @returns {Object} Stream stdout.
 19   */
 20   out: function() { return process.stdout; },
 21   
 22   /**
 23     Returns a new string buffer. Since these methods may be called
 24     asynchronously we can never be sure who might be using a shared
 25     buffer so we're relying on the GC to pick-up after us.
 26     
 27     @returns {Object} A new string buffer.
 28   */
 29   buff: function() {
 30     return xt.string.buffer.create();
 31   },
 32   
 33   /**
 34     For general logging to console use this method. It takes
 35     any number of arguments and attempts to push them to the
 36     output buffer in some meaningful way. For objects it uses
 37     any `toString` method available or does not print the object.
 38     For debugging output (for inspection of objects) and data that
 39     should not display during production use `debug`.
 40     
 41     @todo This description is incomplete and so is the implementation
 42       as there will be an actual logger that is capable of logging to
 43       remote filesystems or local with log management etc.
 44     
 45     @see xt#io#debug
 46     @see xt#io#dev
 47     @see xt#io#err
 48     @see xt#io#warn
 49     
 50     @param {Object} [args] Any number of arguments of any type.
 51   */
 52   log: function log() {
 53     var a = args(),
 54         b = xt.io.buff();
 55     b .set('color', 'grey')
 56       .set('prefix', '<<LOG>>');
 57     a.unshift(b);
 58     xt.io.__console__.apply(this, a);   
 59   },
 60   
 61   /** @public
 62   */
 63   warn: function warn() {
 64     var a = args(),
 65         b = xt.io.buff();
 66     b .set('color', 'yellow')
 67       .set('prefix', '<<WARN>>');
 68     a.unshift(b);
 69     xt.io.__console__.apply(this, a);
 70   },
 71   
 72   /** @public
 73   */
 74   dev: function dev() {
 75     var a = args(),
 76         b = xt.io.buff();
 77     b .set('color', 'blue')
 78       .set('prefix', '<<DEV>>');
 79     a.unshift(b);
 80     xt.io.__console__.apply(this, a);    
 81   },
 82   
 83   /** @public
 84   */
 85   err: function err() {
 86     var a = args(),
 87         b = xt.io.buff();
 88     b .set('color', 'red')
 89       .set('prefix', '<<ERROR>>');
 90     a.unshift(b);
 91     xt.io.__console__.apply(this, a);
 92   },
 93   
 94   /** @public
 95   */
 96   debug: function debug() {
 97     var a = args(),
 98         b = xt.io.buff();
 99     b .set('color', 'blue')
100       .set('prefix', '<<DEBUG>>');
101     a.unshift(b);
102     xt.io.__console__.apply(this, a);
103   },
104   
105   /**
106     Reusable portion of code for output.
107     
108     @param {Object} the string buffer to use
109     @param {Object...} any objects to output in the buffer
110     @private
111   */
112   __console__: function __console__() {
113     if(arguments.length < 2) return;
114     var a = args(),
115         b = a.shift(),
116         l = a.length,
117         i = 0;
118     for(; i<l; ++i)
119       b.push(a[i]);
120     console.log(b.flush());
121   }
122 }
