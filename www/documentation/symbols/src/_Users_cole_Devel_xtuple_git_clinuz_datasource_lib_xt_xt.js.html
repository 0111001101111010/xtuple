 21 /** 
 22   A bootstrapper for the server and services once all of the
 23   third-party modules have been added to the namespace and are
 24   available as well as the required components of xt in its
 25   foundation are also loaded and available.
 26 
 27   @lends xt
 28   @private
 29 */
 30 xt.bootstrap = function bootstrap() {
 31   
 32   // special case where the desired output requires calling console directly
 33   xt.io.__console__(xt.string.buffer.create({ color: 'blue', prefix: null }),
 34     "\n================================================" +
 35     "\nXTUPLE NODE.JS SERVER FRAMEWORK ({version})".f({ version: xt.version }) +
 36     "\n================================================" +
 37     "\nThis server is highly experimental - it is an alpha." +
 38     "\n\nPlease report bugs to the project git issue tracker and for blocking" +
 39     "\nissues please also report those via email to Cole Davis (cole@xtuple.com)\n"
 40   );
 41   
 42   // let any objects that are waiting know that we are bootstrapping
 43   process.emit('xtBootstrapping');
 44 
 45   // we can start a server once all of the routers/routes are
 46   // are loaded and available
 47   process.once('xtServerReady', function() {
 48     xt.server.create({
 49       name: 'dataServer',
 50       port: xt.opts['server-port'],
 51       autoStart: YES,
 52       router: xt.dataRouter
 53     });
 54     
 55     // sigh...
 56     process.emit('xtBootstrapped');
 57   });
 58 }
