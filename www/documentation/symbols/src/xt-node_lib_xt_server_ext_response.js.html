  2 /** @namespace
  3   The xt response object provides a mechanism to pass around
  4   data from client requests of various types and have methods interact
  5   and be able to respond via a single API. It provides a means
  6   of inspection of the original server request/response objects
  7   but also a means to push back various types of data such as
  8   JSON payloads or bitstreams.
  9 */
 10 xt.response = xt.object.extend(
 11   /** @lends xt.response.prototype */ {
 12 
 13   /**
 14     Initialization of the response object ends in a call back to
 15     the router that created it emitting an event to actually
 16     route the request once any payload has been acquired.
 17     
 18     @private
 19   */
 20   init: function() {
 21     var req = this.get('request'),
 22         r = this.get('router'),
 23         s = this; 
 24 
 25     // make sure to retrieve any payload data
 26     req.on('data', function(c) {
 27       s.payloadData += c;
 28     }).on('end', function() {
 29       s.consume();
 30       r.emit('xtrReady', s);
 31     });
 32   },
 33 
 34   /**
 35     Any payload data retrieved from the original
 36     server request object on initialization.
 37 
 38     @property
 39   */
 40   payloadData: '',
 41 
 42   /**
 43     The server request object.
 44 
 45     @property
 46   */
 47   request: null,
 48 
 49   /**
 50     The server response object.
 51 
 52     @property
 53   */
 54   response: null,
 55 
 56   /**
 57     The server that owns the router that owns this
 58     response object.
 59 
 60     @property
 61   */
 62   server: null,
 63 
 64   /**
 65     The router that owns this response object.
 66 
 67     @property
 68   */
 69   router: null,
 70 
 71   /**
 72     If the server response object can still be written to.
 73     Will be YES unless the pipe has been closed. Subsequent
 74     requests to close the connection when this is already
 75     NO will be ignored.
 76 
 77     @type Boolean
 78     @default YES
 79     @property
 80   */
 81   isOpen: YES,
 82 
 83   /**
 84     Explicit write to response stream. If no content-type
 85     or additional headers have been set this forces
 86     implicit headers.
 87   */
 88   write: function() {
 89     var a = args(),
 90         i = 0,
 91         l = a.length,
 92         r = this.get('response');
 93     for(; i<l; ++i)
 94         r.write(a[i]);
 95     return this;
 96   },
 97 
 98   /**
 99     Close the connection to the client...forcefully.
100   */
101   close: function() {
102     if(this.get('isOpen') === NO) return;
103     this.get('response').end();
104     this.set('isOpen', NO);
105   },
106 
107   /**
108     Consumes information about the server request...
109   */
110   consume: function() {
111     var req = this.get('request'),
112         res = this.get('response'),
113         json = xt.json(this.get('payloadData'));
114   
115     // console.log(xt.util.inspect(res, true, 6));
116 
117     // interesting information about a request that we
118     // store for possible useful reasons and debugging
119     // (because apparently debugging isn't useful...)
120     var i = {
121       url: req.url || null,
122       method: req.method || null,
123       status: req.statusCode || null,
124       clientAddress: req.client.remoteAddress || null,
125       clientPort: req.client.remotePort || null,
126       server: this.get('server.name') || null,
127       serverPort: req.client.address().port || null,
128       serverAddress: req.client.address().address || null,
129       payload: this.get('payloadData') || null,
130       json: json || null
131     },
132 
133     d = "Client request received with the following information",
134     t = "REQUEST INFORMATION";
135 
136     // if there is a leading / strip it
137     if(i.url.indexOf('/') == 0)
138       i.url = i.url.slice(1);
139 
140     this.set('route', i.url);
141 
142     // go and and add these properties to the xt response
143     // object for later if necessary
144     this.set('info', _.clone(i));
145 
146     xt.report(i, t, d);
147   },
148 
149   className: 'xt.response'
150 
151 });
