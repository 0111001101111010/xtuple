  2 /** @namespace
  3   Routes provide custom handling for specific paths designated as 
  4   valid requests from clients. They have custom implementations of
  5   handlers for the generated xt response objects.
  6 */
  7 xt.route = xt.object.extend(
  8   /** @lends xt.route.prototype */ {
  9 
 10   /**
 11     If a router matches a requested path to this route it will call this
 12     method on the route and pass it the generated xt response object. Routes
 13     have custom implementations depending on their purpose so the request
 14     should be specific to the type of route handler.
 15 
 16     @param {Object} xtr The xt response object.
 17     
 18     @method
 19   */
 20   handle: function(xtr) {},
 21 
 22   /**
 23     Routes have a list of the paths they handle. For now the paths are
 24     hard-coded for efficient matching but can/should be extended in the
 25     future to be more versatile.
 26 
 27     @property
 28   */
 29   routes: [],
 30 
 31 });
 32 
 33 xt.mixin(xt.route,
 34   /** @lends xt.route */ {
 35 
 36   loadRoutes: function() {
 37 
 38     xt.log("Loading available routes (from {dir})".f({ dir: 
 39       xt.fs.shorten(xt.fs.normalize(__dirname + '/../routes'), 5) }));
 40 
 41     // chdir to modules directory for convenient
 42     // relative pathing
 43     process.chdir(__dirname);
 44 
 45     // find all of the available files in the routes
 46     // directory
 47     xt.fs.directoryFiles('../routes', function(files) {
 48 
 49       var num = files.length;
 50 
 51       // this might should throw a fatal as there really isn't much that
 52       // can be done when no routes are available
 53       if(num <= 0)
 54         return issue(xt.warning("There are no routes available to the server."));
 55 
 56       // iterate over the files and require them so they can be executed
 57       _.each(files, function(v, k) {
 58         require('../routes/%@'.f(v));
 59         --num; if(num == 0) process.emit('xtRoutesLoaded');
 60       }, this);
 61 
 62     }, 'js');
 63 
 64   }
 65 
 66 });
