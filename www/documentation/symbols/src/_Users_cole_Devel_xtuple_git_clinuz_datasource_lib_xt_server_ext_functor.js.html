  2 /** @class
  3   Functors are special handlers for data requests from the client.
  4   They are only used by the dataRoute xt route object and are loaded
  5   automatically at server start.
  6 */
  7 xt.functor = xt.object.extend(
  8   /** @lends xt.functor.prototype */ {
  9 
 10   /**
 11     When loaded by the data route handler it is auto-created
 12     and thus we want to auto-register to be selectable by
 13     the route.
 14 
 15     @private
 16   */
 17   init: function() {
 18     if(xt.none(xt.functors)) xt.functors = [];
 19     xt.functors.push({ handles: this.get('handles'), target: this });
 20   },
 21 
 22   /**
 23     A functor must provide a handle method that takes
 24     a single parameter: the xt response obejct. From here
 25     it has access to the original response and request objects
 26     as well as the payload data. The session object will be
 27     attached to the xt response object by the time this
 28     method is executed when one can be present. If a functor
 29     requires a session but the sessionStore fails to retrieve
 30     the session this method will not be executed.
 31 
 32     @param {Object} xtr The xt response object.
 33   */
 34   handle: function(xtr) {},
 35 
 36   /**
 37     Each functor handles a specific request type from
 38     the client. This is a string representing the request
 39     type that it should be selected when this type is
 40     encountered by the server.
 41 
 42     @type {String}
 43     @property
 44   */
 45   handles: null,
 46   
 47   /**
 48     Almost all functors should require a session. If they do,
 49     they will automatically attempt to acquire the session before
 50     handling the request to ensure they can execute their
 51     functionality as the correct user. In the rare cases that
 52     a functor does not require an active session (such as a request
 53     for a new session) this boolean must be set to NO but it
 54     defaults to YES.
 55     
 56     @property
 57     @type Boolean
 58     @default YES
 59   */
 60   needSession: YES,
 61 
 62   /** @private */
 63   className: 'xt.functor'
 64 
 65 });
