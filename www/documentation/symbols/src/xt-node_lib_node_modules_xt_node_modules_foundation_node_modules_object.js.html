  2 /** 
  3   @class 
  4   
  5   The xt.object class is the base class for nearly all custom
  6   objects in the xt framework for node.js server.
  7   
  8   @param {Object} [hashes] Hashes or a hash to extend from.
  9   @returns {Object} A new instance of an xt.object or child of xt.object.
 10 */
 11 xt.object = function() { 
 12   this.__super__ = xt.object.prototype;
 13   return xt.__init__.apply(this, arguments[0]); 
 14 }

 16 //.............................................
 17 // Class Extensions
 18 //
 19 // These methods make using xt.object's more convenient and allow
 20 // for easy extensibility and inheritance.
 21 xt.mixin(xt.object,
 22   /** @lends xt.object */ {
 23   
 24   /**
 25     Creates a new object based on this object type. Accepts hashes
 26     for extension that will create either a new class type as a singleton
 27     or simply a new object of this type with properties initialized to
 28     the given values.
 29     
 30     @param {Object} [hashes] Mixins to be added to the object.
 31     @returns {Object} An instance of the [extended] object.
 32   */
 33   create: function create() { 
 34     var c = this, r = new c(arguments); return r; 
 35   },
 36   
 37   /**
 38     Extends the base object with the given properties. It creates a new class
 39     that inherits from the base. Can take any number of key-value hashes to
 40     extend the base with that redefine it as a new class or simply the same base
 41     class with different properties at initialization.
 42     
 43     @param {Object} [hashes] Mixins to be added to the object.
 44     @returns {Object} An extended prototype.
 45   */
 46   extend: function extend() {
 47     var r = function() { return xt.__init__.apply(this, arguments[0]); },
 48         l = arguments.length,
 49         i = 0, k, p;
 50     for(k in this)
 51       if(!this.hasOwnProperty(k)) continue;
 52       else r[k] = this[k];
 53     p = (r.prototype = xt.__sire__(this.prototype));
 54     for(; i<l; ++i)
 55       xt.__proto_extend__(p, arguments[i], this.prototype);
 56     p.constructor = r;
 57     p.__super__ = this.prototype;
 58     return r;
 59   },
 60   
 61   /**
 62     Simple mechanism to differentiate between instance and class.
 63     
 64     @return {String} Formatted name of the class.
 65   */
 66   toString: function() { 
 67     var c = this.className
 68       || this.prototype.className;
 69     return "[ CLASS: %@ ]".format(c);
 70   }
 71   
 72 }) ;

 74 /** @lends xt.object.prototype */
 75 xt.object.prototype = {
 76   
 77   /** @private
 78     
 79     @todo This has things to do...
 80   */
 81   init: function() {},
 82   
 83   /**
 84     Set a property on the given object to be equal to a specified
 85     value. This is essentially meta-data of the object and can be
 86     referenced and manipulated many ways but also allows for types
 87     of notifications to take place.
 88     
 89     @param {String} key The key/path for the property to set (relative).
 90     @param {Object} value The object (any type) to place at path.
 91     @returns {Object} Callee for chaining.
 92   */
 93   set: function() {
 94     return xt.__set__.apply(this, arguments);
 95   },
 96   
 97   /**
 98     Retrieve a value for the given property path on the object.
 99     
100     @param {String} key The key to lookup and retrieve.
101     @returns {Object|Boolean} The value for the key or undefined if not found.
102   */
103   get: function() {
104     return xt.__get__.apply(this, arguments);
105   },
106   
107   /**
108     Each class should have a name. This name is used in several
109     ways of matching types, retrieving classes from strings,
110     comparisons, etc.
111 
112     @private
113   */
114   className: 'xt.object',
115   
116   /**
117     While this method can be overridden, most cases it won't be.
118     It provides a consistent, accurate and relatively convenient
119     human-readable output for debugging (mostly) but sometimes
120     comparisons.
121     
122     @returns {String} The string for the object.
123     @private
124   */
125   toString: function toString() {
126     var c = this.className || 'NOCLASS',
127         g = this.uid || 'NOUID';
128     return "[ %@ (%@) ]".format(c, g);
129   }
130 }

132 // We want for xt.objects to inherit from EventEmitter so they can
133 // automatically handle events. Obviously there is a tradeoff by
134 // supplying it to all xt.objects but it appears to handle things
135 // well enough.
