 26   /**
 27     Find and return paths to all files in a specified directory.
 28     Optionally an extension can be specified as the third paramater.
 29     The second parameter is the callback on completion. This method is
 30     asynchronous.
 31   
 32     @param {String} path The path to the directory.
 33     @param {Function} callback The callback expects a single (array) parameters of the results.
 34     @param {String} [extension] The file extension to look for (otherwise all file types).
 35   */
 36   directoryFiles: function(p, c, t) {
 37     p = this.__path__.normalize(p);
 38     try {
 39       this.__fs__.readdir(p, function(e, f) {
 40         c(xt.typeOf(t) === xt.t_string ? xt.fs.reduce(f, t) : f);
 41       })
 42     } catch(e) { 
 43       xt.warn(e); 
 44       c([]); 
 45     }
 46   }, 
 47 
 48   /**
 49     See Node.js documentation. Reimplements path.normalize.
 50   */
 51   normalize: function(path) {
 52     return this.__path__.normalize(path);
 53   },

 55   /**
 56     Takes what might be a very long file path and shortens it to
 57     a (default 3) maximum number of slashes counting from the end
 58     and prepends an elipses. This is strictly for aesthetic/human-
 59     readability purposes.
 60 
 61     @param {String} path The file path needing to be shortended.
 62     @param {Number} [max] The maximum number of slashes to include.
 63     @returns {String} The normalized/shortened file path.
 64   */
 65   shorten: function(path, max) {
 66     if(xt.none(max)) max = 3;
 67     if(xt.none(path) || xt.typeOf(path) !== xt.t_string) return path;
 68 
 69     var c = path.match(/\//g).length;
 70 
 71     if(!c || c <= max) return path;
 72     
 73     var f = 0,
 74         d = c - max,
 75         i = 0, t = path;
 76 
 77     while(f < d && !!~i) {
 78       i = t.indexOf('/');
 79       t = t.slice(i == 0 ? 1 : i);
 80       if(i != 0) f += 1;
 81     }
 82 
 83     return '...{path}'.f({ path: t });
 84   },

 86   /**
 87     Reads the contents of a given file and/or path. This assumes utf-8
 88     encoding and is asynchronous in execution.
 89     
 90     @param {String} path The path to the directory containing the file
 91       or the full path to the file.
 92     @param {String} filename The name of the file if only directory path is provided (will be normalized).
 93     @param {Function} callback The callback takes an error and filedata paramaters.
 94   */
 95   readFile: function(p, f, c) {
 96     if(xt.typeOf(f) === xt.t_function) {
 97       c = f;
 98       f = '';
 99     }
100     p = this.__path__.join(p, f);
101     try {
102       var d = this.__fs__.readFile(p, 'utf-8', c)
103     } catch(e) {
104       xt.warn(e);
105       c("");
106     }
107   },
108   
109   /**
110     Splits a string into individual lines. Can be used for normal strings or
111     from file data. This function is synchronous and returns an array of lines.
112   
113     @param {String} string The string or file data to parse.
114     @returns {Array} The array of lines from the string.
115   */
116   byLine: function(f) {
117     if(!f || f.length == 0 || xt.typeOf(f) !== xt.t_string) return [];
118     return f.split('\n') || [];
119   },

121   /**
122     Reduces an array of filenames to just files with the specified
123     extension. This is a synchronous function.
124     
125     @param {Array} files The array of filenames.
126     @param {String} extension The filetype extension to look for.
127     @returns {Array} The array of files with the provided extension.
128   */
129   reduce: function(f, t) {
130     if(!f || f.length < 1) return f || [];
131     var i = 0,
132         l = f.length,
133         n = [],
134         s = t.length, p;
135     for(; i<l; ++i) {
136       p = f[i].length - s;
137       if(!!~f[i].indexOf(t) && f[i][p] == t[0]) n.push(f[i]);
138     }
139     return n;
140   }
