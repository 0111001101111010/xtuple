1 /** 
  2   @class
  3 
  4   Object for simplified string-buffer-like operations. Really
  5   for convenience only as it has some overhead.
  6 
  7   @example
  8 
  9     Example usage
 10     
 11     var buf = xt.string.buffer.create(); // or xt.io.buff() for same result
 12     buf.push("one string", "two string", "three string");
 13     buf.set("color", "blue");
 14     buf.set("prefix", "SAMPLE >> ");
 15     
 16     var out = buf.flush(); // => `SAMPLE >> one string two string three string`
 17 
 18 */
 19 xt.string.buffer = xt.object.extend(
 20   /** @lends xt.string.buffer.prototype */ {
 21 
 22   /**
 23     Push a variable number of string arguments into the buffer.
 24 
 25     @param {String} [strings] String(s) to be added to the buffer.
 26     @return {Object} Callee for chaining.
 27   */
 28   push: function() {
 29     if(!this.buffer || xt.typeOf(this.buffer) !== xt.t_array)
 30       this.buffer = [];
 31     var a = args(),
 32         i = 0,
 33         l = a.length;
 34     for(; i<l; ++i) this.buffer.push(a[i]);
 35     return this;
 36   },
 37 
 38   /**
 39     Concatenate strings in the buffer and return the formatted
 40     string.
 41 
 42     @returns {String} The concatenated, formatted string from the
 43       buffer contents.
 44   */
 45   flush: function() {
 46     var b = this.buffer || [],
 47         i = 0,
 48         l = b.length, 
 49         p = this.get('prefix'),
 50         s = p ? p.trim() + ' ' : '';
 51     s = xt.colors[this.color](s);
 52     for(; i<l; ++i) {
 53       if(xt.typeOf(b[i]) === xt.t_string)
 54         s += xt.colors[this.color](b[i]);
 55       else s += ' ' + 
 56         xt.colors.grey(xt.util.inspect(b[i], NO, 3));
 57       // if(i+1 != l) s += ' ';
 58     }
 59     this.buffer = null;
 60     return s;
 61   },
 62 
 63   /**
 64     The color to produce on color-capable terminal displays.
 65     @default grey
 66     @property
 67   */
 68   color: 'grey',
 69 
 70   /** 
 71     The tag to display before the output to the terminal.
 72     @default Log:
 73     @property 
 74   */
 75   prefix: 'Log:',
 76 
 77   /** @private */
 78   className: 'xt.string.buffer'
 79 });
