  6 /** 
  7   A bootstrapper for the server and services once all of the
  8   third-party modules have been added to the namespace and are
  9   available as well as the required components of xt in its
 10   foundation are also loaded and available.
 11 
 12   @lends xt
 13   @private
 14 */
 15 xt.bootstrap = function bootstrap() {
 16   
 17   // special case where the desired output requires calling console directly
 18   xt.io.__console__(xt.string.buffer.create({ color: 'blue', prefix: null }),
 19     "\n================================================" +
 20     "\nXTUPLE NODE.JS SERVER FRAMEWORK ({version})".f({ version: xt.version }) +
 21     "\n================================================" +
 22     "\nThis server is highly experimental - it is an alpha." +
 23     "\n\nPlease report bugs to the project git issue tracker and for blocking" +
 24     "\nissues please also report those via email to Cole Davis (cole@xtuple.com)\n"
 25   );
 26   
 27   // first thing to do is test to make sure we can communicate with the database
 28   xt.database.check();  
 29 
 30   // need to load any available routes for routers to be able to do anything
 31   // with incoming traffic
 32   xt.route.loadRoutes();
 33 
 34   process.once('xtRoutesLoaded', function() {
 35 
 36     // need to load any available routers for servers to be able to name them
 37     // explicitly
 38     xt.router.loadRouters();
 39 
 40     process.once('xtRoutersLoaded', function() {
 41       xt.server.create({
 42         name: 'dataServer',
 43         port: xt.opts['server-port'],
 44         autoStart: YES,
 45         router: xt.dataRouter
 46       });
 47     });
 48   });
 49 
 50 }
