  9 /** @namespace
 10   The server if a fundamental object that can be created/instanced to
 11   server various purposes. It also has some convenience features
 12   and static methods as helpers.
 13 */
 14 xt.server = xt.object.extend(
 15   /** @lends xt.server.prototype */ {
 16   
 17   /**
 18     If this is set to YES then the server
 19     will automatically attempt to start itself upon
 20     being created.
 21 
 22     @type {Boolean}
 23     @default NO
 24   */
 25   autoStart: NO,
 26 
 27   /**
 28     The port for the server to bind to.
 29 
 30     @type {Number}
 31     @default null
 32   */
 33   port: null,
 34 
 35   /**
 36     Human-readable identifier for the server. Will be
 37     randomly generated if one is not provided. If one
 38     is provided it can be used to grab a reference
 39     to the server once it has been created.
 40 
 41     @type {String}
 42     @default null
 43   */
 44   name: null,
 45 
 46   /**
 47     The router for the server to use when it receives
 48     requests. If none is provided it will use a default
 49     router.
 50 
 51     @type {Router}
 52     @default xt.router_default
 53   */
 54   router: xt.router_default,
 55   
 56   /**
 57     The HTTP server reference created once the xt.server has
 58     been instanced.
 59 
 60     @private
 61   */
 62   _server: null,
 63 
 64   /**
 65     During initialization it the server determines the appropriate
 66     properties and then waits to be started (or auto-starts if
 67     autoStart is set to true).
 68     
 69     @private
 70   */
 71   init: function() {
 72     var p = this.get('port'),
 73         r = this.get('router'),
 74         a = this.get('autoStart'),
 75         n = this.get('name') || xt.server.name();
 76     if(xt.none(p))
 77       throw xt.fatal("Could not create server {name}, no port provided".f(
 78         { name: n }));
 79     if(xt.none(r))
 80       throw xt.fatal("Could not create server {name}, router is invalid".f(
 81         { name: n }));
 82     else
 83       r.set('server', this);
 84     if(a) this.start();
 85   },
 86 
 87   /**
 88     Starts the server with the properties set on it.
 89 
 90     @method
 91     @returns Callee
 92   */
 93   start: function() {
 94     var s = this,
 95         p = this.get('port');
 96     try {
 97       s._server = xt.http.createServer(
 98         function(req, res) {
 99           var r = s.get('router'); // allows swapping of routers
100                                    // is that good?
101           r.handle.call(r, req, res); 
102         }
103       ).listen(p);
104     } catch(e) { xt.exception.handle(xt.fatal(e)); }
105 
106     xt.log("Started server, {name}, listening on port {port}".f(
107       { name: s.get('name'), port: p }));
108 
109     return this;
110   }
111 
112 });
