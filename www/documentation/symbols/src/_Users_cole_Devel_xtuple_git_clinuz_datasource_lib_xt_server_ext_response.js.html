  2 /** @namespace
  3   The xt response object provides a mechanism to pass around
  4   data from client requests of various types and have methods interact
  5   and be able to respond via a single API. It provides a means
  6   of inspection of the original server request/response objects
  7   but also a means to push back various types of data such as
  8   JSON payloads or bitstreams.
  9 */
 10 xt.response = xt.object.extend(
 11   /** @lends xt.response.prototype */ {
 12 
 13   /**
 14     Initialization of the response object ends in a call back to
 15     the router that created it emitting an event to actually
 16     route the request once any payload has been acquired.
 17     
 18     @private
 19   */
 20   init: function() {
 21     var req = this.get('request'),
 22         r = this.get('router'),
 23         s = this; 
 24 
 25     // for development may need to create fake
 26     // requests so allow them to not always have
 27     // legit requests...
 28     if(xt.none(req))
 29       return this.set('isOpen', NO);
 30     
 31     // make sure to retrieve any payload data
 32     req.on('data', function(c) {
 33       s.payloadData += c;
 34     }).on('end', function() {
 35       s.consume();
 36       r.route(s);
 37     });
 38   },
 39 
 40   /**
 41     Any payload data retrieved from the original
 42     server request object on initialization.
 43 
 44     @property
 45     @type String
 46   */
 47   payloadData: '',
 48 
 49   /**
 50     Any payload data retrieved from the original
 51     server request object on initialization but
 52     parsed as JSON.
 53     
 54     @property
 55     @type Object
 56   */
 57   payloadJSON: null,
 58   
 59   /**
 60     The database expects keys to have been decamelized.
 61     This is the decamelized and database-safe/friendly object.
 62     
 63     @property
 64     @type Object
 65   */
 66   payloadSafe: null,
 67 
 68   /**
 69     If the current request is part of a valid session.
 70     
 71     @see xt#response#session
 72     @property
 73     @default NO
 74     @type Boolean
 75   */
 76   useSession: NO,
 77 
 78   /**
 79     The server request object.
 80 
 81     @property
 82     @type Object
 83   */
 84   request: null,
 85 
 86   /**
 87     The server response object.
 88 
 89     @property
 90     @type Object
 91   */
 92   response: null,
 93 
 94   /**
 95     The server that owns the router that owns this
 96     response object.
 97 
 98     @property
 99     @type Object
100   */
101   server: null,
102 
103   /**
104     The router that owns this response object.
105 
106     @property
107     @type Object
108   */
109   router: null,
110 
111   /**
112     If the server response object can still be written to.
113     Will be YES unless the pipe has been closed. Subsequent
114     requests to close the connection when this is already
115     NO will be ignored.
116 
117     @default YES
118     @property
119     @type Boolean
120   */
121   isOpen: YES,
122   
123   /**
124     On occassions when a session is needed by a functor,
125     this will be a reference to the functor to execute
126     once the session has been successfully acquired.
127     
128     @property
129     @type xt.functor
130     @default null
131   */
132   handler: null,
133 
134   /**
135     Explicit write to response stream. If no content-type
136     or additional headers have been set this forces
137     implicit headers.
138   */
139   write: function() {
140     var a = args(),
141         i = 0,
142         l = a.length,
143         r = this.get('response');
144         
145     for(; i<l; ++i) {     
146       if(xt.typeOf(a[i]) === xt.t_hash)
147         a[i] = xt.json(a[i]) || ''; // we need to be able to know this failed
148       //xt.debug("xt.response (%@) writing: %@".f(this.toString(), a[i]));
149       r.write(a[i]);
150     }
151     return this;
152   },
153 
154   /**
155     Close the connection to the client...forcefully.
156   */
157   close: function() {
158     if(this.get('isOpen') === NO) return;
159     this.get('response').end();
160     this.set('isOpen', NO);
161   },
162 
163   /**
164     Consumes information about the server request...
165   */
166   consume: function() {
167     
168     
169   //   xt.debug("in xt.response.consume");
170     
171     
172     var req = this.get('request'),
173         res = this.get('response'),
174         json = xt.json(this.get('payloadData')),
175         dbsafe = xt.json(this.get('payloadData'));
176         dbsafe = dbsafe.decamelize();
177 
178     this.set('payloadJSON', json);
179     this.set('payloadSafe', xt.json(dbsafe));
180   
181     // interesting information about a request that we
182     // store for possible useful reasons and debugging
183     // (because apparently debugging isn't useful...)
184     var i = {
185       url: req.url || null,
186       method: req.method || null,
187       status: req.statusCode || null,
188       clientAddress: req.client.remoteAddress || null,
189       clientPort: req.client.remotePort || null,
190       server: this.get('server.name') || null,
191       serverPort: req.client.address().port || null,
192       serverAddress: req.client.address().address || null,
193       payload: this.get('payloadData') || null,
194       json: json || null
195     },
196 
197     d = "Client request received with the following information",
198     t = "REQUEST INFORMATION";
199 
200     // if there is a leading / strip it
201     if(i.url.indexOf('/') == 0)
202       i.url = i.url.slice(1);
203 
204     this.set('route', i.url);
205 
206     // go and and add these properties to the xt response
207     // object for later if necessary
208     this.set('info', _.clone(i));
209 
210     xt.report(i, t, d);
211   },
212 
213   getSession: function(p, useSession) {
214     
215     // xt.debug("xt.response.useSession %@".f(useSession ? "YES" : "NO"));
216     
217     if(!useSession) return;
218     
219     xt.sessionStore.getSession(this);
220   }.by('useSession'),
221 
222   className: 'xt.response'
223 
224 });
