  2 /** @class
  3   Session objects are nothing more than context containers. They
  4   are created once a client requests a session and is validated
  5   as a current/active user. Active sessions enable the queries
  6   to be executed as the correct database user (for privilege
  7   enforcement), tracking and additional automated events that
  8   use the current user (triggers).
  9 
 10   <p></p>
 11 
 12   Sessions are fairly dumb objects that are acted on by the
 13   session store. They have a limited API and automatically
 14   generate query prefixes for the user assigned to them while
 15   also supplying a special callback during queries. Queries
 16   are executed under the context of the session to which they
 17   belong and this gives access to the xt response object
 18   supplied during client requests and allows for the callback
 19   to write to the correct client.
 20 
 21   <p></p>
 22 
 23   **WARNING**
 24 
 25   <p></p>
 26 
 27   In this pass (as of 01/24/2012) sessions are very unsecure.
 28   An effort will be made to clean up this process but as such
 29   they are volatile and the API is expected to change
 30   dramatically in the future.
 31 
 32   @example
 33     
 34     // this is unrealistic because sessions are created
 35     // automatically on client requests not manually but
 36     // for the sake of example...
 37     var s = xt.session.create({ 
 38       info: { user: 'cole', ip: '127.0.0.1' }
 39     });
 40 
 41     // note that the query will be prefixed properly
 42     // automatically to ensure it is executed in the correct
 43     // context of the appropriate user
 44     // also note that error handling is already dealt with
 45     // by the session object so the callback will only
 46     // be executed when errors do not exist
 47     s.query('select * from public.incdt', function(r) {
 48 
 49       // inside the callback we are in the context of the
 50       // session and can request the xt response object (xtr)
 51       var xtr = this.get('xtr');
 52       
 53       // write some response from the successfully executed
 54       // query directly back to the client as the payload
 55       // of the response and close the response object
 56       xtr.write(xt.json(r)).close();
 57     });
 58 */
 59 xt.session = xt.object.extend(
 60   /** @lends xt.session.prototype */ {
 61 
 62   /**
 63     Allows queries to be run correctly according to their
 64     active session information. If the callback function
 65     is not provided the open server response object will
 66     automatically be closed with no bytes written to the
 67     client. The callback will be executed within the context
 68     of the session object and allows the use of this inside
 69     to retrieve the xt response object or other properties.
 70 
 71     @param {String} query The query string to execute under the session.
 72     @param {xt.response} xtr The xt response object.
 73     @param {Function} [func] The callback for completed queries.
 74     @method
 75   */
 76   query: function(query, xtr, func) {
 77     if(!this.get('isActive')) return issue(xt.close(
 78       "Could not execute query, invalid or inactive session", this.get('xtr')));
 79     this.set('xtr', xtr);
 80     query = query.pre(this._xt_userprefix);
 81     if(xt.none(func) || xt.typeOf(func) !== xt.t_func)
 82       func = xt.session._xt_default_callback;
 83 
 84     var s = this,
 85         f = function(e, r) {
 86           if(e)
 87             issue(xt.close("Error during session query, %@".f(e), s.get('xtr')));
 88           else func.apply(s, r);
 89           s.set('xtr', null);  
 90         };
 91 
 92     // ...
 93     xt.db.query(query, f);
 94   }, 
 95 
 96   /**
 97     Returns whether or not the current session is still active.
 98 
 99     @returns {Boolean} YES|NO on session active status.
100   */
101   isActive: function() {
102     return xt.sessionStore.validate(this);
103   }.property(),
104 
105   /**
106     Initializes the xt session object with the
107     data passed to it at creation.
108 
109     @private
110   */
111   init: function() {
112     var n = this.get('info'),
113         u = n.user,
114         i = n.ip,
115         t = xt.session.time(),
116         p = xt.session._xt_seteffectiveuser;
117     this.set('user', u);
118     this.set('ip', i);
119     this.set('started', t);
120     this.set('lastUsed', t);
121     this._xt_userprefix = p.f({ user: u });
122   },
123 
124   /** @private */
125   className: 'xt.session'
126 
127 });
128 
129 xt.mixin(xt.session,
130   /** @lends xt.session */ {
131 
132   /**
133     The default template query string to be prepended to
134     additional queries under this session.
135     
136     @property
137     @private
138   */
139   _xt_seteffectiveuser: "select seteffectiveuser('{user}'); ",
140 
141   /**
142     Returns the current UTC time.
143 
144     @returns {Integer} The UTC current time in milliseconds.
145   */
146   time: function() {
147 
148     // this seems quite wasteful...
149     return new Date().UTC();
150   },
151 
152   /**
153     Returns a unique hash for a particular session.
154 
155     @returns {String} The hash for the session.
156     @method
157   */
158   hash: function() {
159     return xt.crypto
160       .createHash('md5')
161       .update(Math.random().toString())
162       .digest('hex');
163   },
164 
165   /**
166     The default callback for queries where a callback is not
167     supplied. Simply closes the xt response to the client.
168 
169     @method
170   */
171   _xt_default_callback: function(e, r) {
172     if(e) return issue(
173       xt.close("Error during query, %@".f(xt.inspect(e)), this.get('xtr')));
174     r.close();
175   }
176 
177 });
