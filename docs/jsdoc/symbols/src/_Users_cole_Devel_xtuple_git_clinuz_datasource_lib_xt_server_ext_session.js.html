  4 xt.session = xt.object.extend(
  5   /** @lends xt.session.prototype */ {
  6    
  7   /** @private */
  8   init: function() {
  9     
 10     // xt.debug("in xt.session.init");
 11     
 12     var xtr = this.get('xtr'),
 13         u = this.get('userName'),
 14         k = this.get('userKey'),
 15         d = this.get('data'),
 16         s = d ? d.sid || '' : '',
 17         // s = this.get('sid'),
 18         h = xtr.get('handler'), // if a handler is waiting on the session
 19         t = {},
 20         i = xtr.get('info'),
 21         self = this;
 22     
 23     
 24     t.userKey = k;
 25     t.sid = s = s || xt.session.id();
 26     t.userName = u;
 27     t.ip = i.clientAddress;
 28     
 29     this.set('sid', s);
 30     
 31     // make sure the xt response knows it has a session
 32     xtr.set('session', this);
 33      
 34     
 35     
 36     if(xtr.get('useSession') && !xt.none(h))
 37       this.once('xtReady', function() { h.handle(xtr); });
 38      
 39      
 40     this.__setup__(xt.none(d) ? {} : d, t);
 41   },
 42   
 43   /**
 44     Commits the current session data collection into the cache.
 45     The data checksum will be updated after the lastModified
 46     property is updated to the current time.
 47     
 48     @param {Function} [callback] The callback to execute after the
 49       write to the cache has transpired.
 50     @method
 51   */
 52   commit: function(cb) {
 53     var d = this.get('data'),
 54         s = this.get('sid'),
 55         self = this, c, key;
 56     cb = cb && xt.typeOf(cb) === xt.t_function ? cb : null;
 57     if(xt.none(d) || xt.none(s)) return;
 58     d.lastModified = xt.session.timestamp();
 59     c = xt.session.checksum(d);
 60     d.checksum = c;
 61     this.set('checksum', c);
 62     this.data = d;
 63     key = '%@:%@:%@'.f(d.userName, d.ip, d.sid);
 64     
 65     var f = function(e, r) {
 66       if(e) return issue(xt.warn("Could not successfully commit session data"));
 67       if(cb && xt.typeOf(cb) === xt.t_function) cb(e, r);
 68       
 69       // in case anyone is interested in when we're done
 70       self.emit('xtReady');
 71     };
 72     
 73     xt.cache.hmset(key, d, f);
 74   },
 75     
 76   reload: function() {},
 77     
 78   
 79   query: function(q, cb) {
 80     q = q.pre("select seteffectivextuser('%@'); ".f(this.get('userName'))),
 81     self = this;
 82     
 83     // xt.debug("xt.session.query => %@".f(q));
 84     
 85     var f = function(e, r) {
 86       
 87       // if there is an error go ahead and just forward it to the
 88       // the callback to let it handle it
 89       if(e) {
 90         if(cb && xt.typeOf(cb) === xt.t_function) return cb(e, r);
 91         else return issue(xt.close("Error in session query but no callback to respond: %@".f(e.message), self.get('xtr')));
 92       }
 93       
 94       // since we know there's going to be additional crap inside
 95       // the response from the session query go ahead and strip it
 96       // transparently and let the handler think everything is
 97       // david bowie...i mean, hunky dory
 98       if(r.rows)
 99         if(r.rows[0].keys().contains('seteffectivextuser'))
100           r.rows.shift();
101       
102       // there, all pretty again
103       if(cb && xt.typeOf(cb) === xt.t_function) cb(e, r);
104     };
105     
106     xt.db.query(q, f);
107   },
108     
109     
110   /** @private */
111   __setup__: function(data, bits) {
112     
113     xt.mixin(data, bits);
114     this.data = data;
115     this.data.created = this.data.created || xt.session.timestamp();
116     
117     // xt.debug("xt.session.__setup__ \n", this.get('data'));
118     
119     this.commit();
120   }
121     
122 });

127 process.once('xtReady', function() {
128   xt.mixin(xt.session, 
129     /** @lends xt.session */ {
130     
131     /**
132       Creates a checksum of the parameter.
133       
134       @param {Object|String} data The data to be digested into a checksum.
135       @returns {String} The checksum of the data.
136       @method
137     */
138     checksum: function(data) {
139       var h = this.__sha1hash__();
140       data = xt.typeOf(data) === xt.t_hash ? xt.json(data) : data;
141       return h.update(data).digest('hex');
142     },
143     
144     
145     /**
146       Creates a unique session id from a random MD5 hash.
147       
148       @returns {String} A unique MD5 session id.
149       @method
150     */
151     id: function() {
152       var h = this.__md5hash__();
153       return h.update(Math.random().toString()).digest('hex');
154     },
155     
156     /**
157       Returns a timestamp in milliseconds.
158       
159       @returns {Number} The current time in milliseconds.
160     */
161     timestamp: function() { return new Date().getTime(); },
162     
163     /**
164       Shared SHA1 hash generator for checksum calculations
165       of the session datasets.
166       
167       @property
168     */
169     __sha1hash__: function() { return xt.crypto.createHash('sha1'); },
170     
171     /**
172       Shared MD5 hash generator for creating *unique*
173       session id's.
174     
175       @property
176     */
177     __md5hash__: function() { return xt.crypto.createHash('md5'); }
178     
179   });
180 });
