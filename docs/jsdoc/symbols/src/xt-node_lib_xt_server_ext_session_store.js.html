  2 /** @class
  3 */
  4 xt.sessionStore = xt.object.create(
  5   /** @lends xt.sessionStore.prototype */ {
  6 
  7   /**
  8     Tests to see if an active session exists for the
  9     passed session id (hash).
 10 
 11     @param {String} sid The session id.
 12     @returns {Boolean} YES|NO if the session does exist.
 13   */
 14   exists: function(sid) {
 15     return !! this.get('sessions')[sid];
 16   },
 17 
 18   /**
 19     Attempts to retrieve an active session object from the
 20     given client data hash.
 21 
 22     @param {Object} info The client information data hash.
 23     @returns {xt.session} The active session object or null if none exists.
 24   */
 25   sessionFromInfo: function(info) {
 26     var i = this.get('ipSessions'),
 27         s = i[info.ip] ? i[info.ip][info.user] : null;
 28     return s; 
 29   },
 30 
 31   /**
 32     Attempts to retrieve an active session object from the
 33     given session id.
 34 
 35     @param {String} sid The session id hash.
 36     @returns {xt.session} The active session object or null if none exists.
 37   */
 38   sessionFromSid: function(sid) {
 39     var s = this.get('sessions');
 40     return s[sid] ? s[sid] : null;
 41   },
 42 
 43   /**
 44     Validates a session. Can take a xt session object or a
 45     session id to determine if the session exists and is
 46     currently valid. Returns false if the session cannot
 47     be found.
 48 
 49     @param {String|xt.session} ses The session id or xt session object.
 50     @returns {Boolean} YES|NO on whether the session is active and valid or not.
 51   */
 52   validate: function(ses) {
 53     if(xt.typeOf(ses) === xt.t_string)
 54       ses = this.sessionFromSid(ses);
 55     if(xt.none(ses)) return NO;
 56     var t = xt.session.time(); 
 57     
 58     // PLACE HOLDER!!!
 59     var MAXDIF = 900000; // 15 minutes for development...
 60 
 61     return Math.abs(t - ses.lastUsed) >= MAXDIF ? NO : YES;
 62   },
 63 
 64   /**
 65     Storage of actual sessions.
 66 
 67     @property
 68   */
 69   sessions: {},
 70 
 71   /**
 72     Redundant storage of sessions by client ip address.
 73     Allows many sessions on the same ip.
 74 
 75     @property
 76   */
 77   ipSessions: {},
 78     
 79   /**
 80     Creates a new session with the given client information.
 81 
 82     @param {Object} info The client information hash.
 83     @returns {xt.session} A xt.session object.
 84   */
 85   createSession: function(info) {
 86     if(xt.none(info)) return NO;
 87     var s = this.get('sessions'),
 88         i = this.get('ipSessions'), n;
 89     n = xt.session.create({
 90       sid: xt.session.hash(),
 91       info: info
 92     });
 93     
 94     if(!i[info.ip]) i[info.ip] = {};
 95     i[info.ip][info.user] = n;
 96 
 97     // add this entry to the sessions store
 98     s[n.sid] = n;
 99 
100     return n;
101   },
102 
103   /**
104     Removes a session from the internal store once it has
105     been deactivated or become inactive.
106 
107     @param {String} sid The session id.
108     @param {Object} info The client information data hash.
109     @returns {Boolean} Returns YES|NO on success/failure respectively.
110   */
111   removeSession: function(sid, info) {
112     var s = this.get('sessions'),
113         i = this.get('ipSessions');
114     s[sid] = null;
115     i[info.ip][info.user] = null;
116     delete s[sid];
117     delete i[info.ip][info.user];
118   },
119 
120   /** @private */
121   className: 'xt.sessionStore'
122 
123 });
