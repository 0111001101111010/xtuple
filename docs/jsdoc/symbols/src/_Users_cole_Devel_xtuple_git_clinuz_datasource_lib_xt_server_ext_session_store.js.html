  3 /** @class
  4   The session store houses the accessor methods
  5   for creating, validating and managing sessions.
  6   Nearly all requests from the client require an
  7   authenticated user with an active session to execute
  8   their requests. 
  9 */
 10 xt.sessionStore = xt.object.create(
 11   /** @lends xt.sessionStore.prototype */ {
 12 
 13   //.......................................
 14   // Public API
 15   //
 16 
 17   /**
 18     Attempts to create a new session for the current
 19     client request and passed in xt response object. If
 20     the user is authenticated then a session will be
 21     generated and the session id will be passed back to
 22     the client. If the user could not be successfully
 23     authenticated the request will be closed.
 24 
 25     @param {xt.response} xtr The xt response object.
 26     @param {Function} [callback] The callback to be executed on success.
 27   */
 28   createSession: function(xtr, cb) {
 29 
 30     // take username, password and ip address
 31     // validate the username and password
 32     // on success continue else fail
 33 
 34 
 35    //  xt.debug("in xt.sessionStore.createSession");
 36 
 37 
 38     var payload = xtr.get('payloadJSON'),
 39         u = payload.username,
 40         p = payload.password,
 41         self = this, q;
 42     q = "select validate_user('%@', '%@')".f(u, p);
 43     xt.db.query(q, function(e, r) {
 44       if(e) return issue(xt.close("Error authenticating user, '%@'".f(u || 'none'), xtr));
 45       r = r.rows[0]; // only looking for single row response from query
 46       if(r.validate_user)
 47         self.loadUser(u, function(e, userkey) {
 48           if(e) return issue(xt.close("Error loading user account form database", xtr));
 49           
 50           // the session will store itself in the cache and will remove
 51           // the reference to the xt response when it is done initializing
 52           // to free it
 53           var session = xt.session.create({ userKey: userkey, xtr: xtr, userName: u });
 54           
 55           if(cb && xt.typeOf(cb) === xt.t_function) session.once('xtReady', function() { cb(session, xtr); });
 56         });
 57       else issue(xt.close("Could not validate user, '%@'".f(u || 'none'), xtr));
 58     });
 59 
 60   },
 61 
 62   /**
 63     For data requests that require an authenticated user with
 64     an active session, each request will pass through this
 65     method. It attempts to validate the user and the session
 66     and if successful will allow the request to continue otherwise
 67     it will close the request with an error to the client.
 68     
 69     @param {xt.response} response The xt response object generated
 70       from the request.
 71     @param {Function} [callback] The callback function to be executed
 72       on succcessful completion of the session validation.
 73     @method
 74   */
 75   validateSession: function(xtr, cb) {
 76 
 77 
 78   //  xt.debug("in xt.sessionStore.validateSession");
 79 
 80     // take username, session id and ip address
 81     // and do a lookup to verify that the session
 82     // exists and is active
 83     
 84     var payload = xtr.get('payloadJSON'),
 85         u = payload.username,
 86         s = payload.sid,
 87         i = xtr.get('info.clientAddress'),
 88         key = '%@:%@:%@'.f(u, i, s);
 89         
 90     // xt.debug("asking cache for %@".f(key));  
 91       
 92     xt.cache.hgetall(key, function(e, r) {
 93       if(e) return issue(xt.close("Error retrieving session data, %@".f(e), xtr));
 94       if(xt.none(r) || r.keys().length <= 0)
 95         return issue(xt.close("Invalid session", xtr));
 96         
 97       var session = xt.session.create({ data: r, xtr: xtr, userName: u, userKey: 'user:%@'.f(u), sid: s });
 98       if(cb && xt.typeOf(cb) === xt.t_function) session.once('xtReady', function() { cb(session, xtr); });
 99     });
100 
101   },
102   
103   /**
104   */
105   getSession: function(xtr) {
106     this.validateSession(xtr);
107   },

113   /**
114     Attempts to load the user into the cache. If the user
115     is already loaded, does nothing but associate the key
116     to the session otherwise it must load the user from
117     the database into the cache and then associate it with
118     the session. The callback is expected to handle error
119     cases.
120 
121     @param {String} username The username to retrieve.
122     @param {Function} callback The callback function to call.
123     @param {Object} [context] The context to execute the callback as.
124     @param {Boolean} [force] Whether or not to force a reload from
125       the database for user data.
126     @method
127   */
128   loadUser: function(u, f, c, force) {
129     if(xt.none(u)) return;
130     if(xt.typeOf(f) !== xt.t_function) return;
131     if(xt.none(c)) c = this;
132     var cb = function(e, r) { f.call(c, e, r); },
133         k = 'user:%@'.f(u);
134     xt.cache.hgetall(k, function(e, r) {
135 
136       // allow the callback to handle error case
137       if(e && !force) return cb(e);
138       if(xt.none(r) || r.keys().length == 0 || force) {
139 
140         xt.debug("Loading user ('%@') from the database into the cache".f(u));
141         
142         // user not loaded into the cache so we need to grab
143         // it from the database and load it into the store
144         var q = "select * from xm.user_account where username='%@'".f(u);
145         xt.db.query(q, function(e, r) {
146 
147           // allow the callback to handle error case
148           if(e) return cb(e);
149           r = r.rows[0];
150 
151           // if there is nothing the user doesn't exist?
152           if(xt.none(r)) return cb(xt.warn("User did not exist ('%@')".f(u)));
153           
154           // we have valid data for the user and we know exactly
155           // what the key will be for them, shove the object into
156           // the cache as a hash and then pass the key into the
157           // callback for success
158           xt.cache.hmset(k, r);
159         });
160       }
161 
162       // all done
163       cb(null, k); 
164     }); 
165   },
166 
167   /**
168   */
169   deleteSession: function() {
170   },

172   /**
173     Automatically set to run at fixed intervals this asynchronous
174     polling method checks for sessions in the cache and determines
175     if they are expired based on server configuration. If the session
176     is expired it will be removed from the cache thus forcing any
177     client attempting to use the session to reauthenticate. This
178     method does not need to be called directly and is managed by
179     the server configuration options.
180     
181     @method
182   */
183   pollCache: function() {
184 
185     // will be set to poll the cache for invalid or
186     // expired sessions and remove them
187     
188     xt.log("Polling cache for expired sessions");
189     
190     var now = xt.session.timestamp(),
191         timeout = xt.sessionStore.sessionTimeout;
192     
193     // first find all of the users that may have sessions and
194     // that are just sitting in the cache
195     xt.cache.keys('user:*', function(e, loadedUsers) {
196       if(e) return issue(xt.warn("Could not poll cache for active users"));
197       var i = 0,
198           l = loadedUsers.length;
199       for(; i<l; ++i) {
200         var user = loadedUsers[i].split(':')[1];
201         
202         // next we need to iterate over them and find any
203         // sessions the cache may have for each user
204         xt.cache.keys('%@:*'.f(user), function(e, sessionsFor) {
205           if(e) return issue(xt.warn("Could not poll cache for active sessions"));
206           var i = 0,
207               l = sessionsFor.length;
208           for(; i<l; ++i) {
209             var sessionKey = sessionsFor[i];
210             
211             // now grab the session data and see if the session is
212             // old enough to remove
213             xt.cache.hgetall(sessionKey, function(e, data) {
214               if(e) return issue(xt.warn("Could not retrieve session data for entry while polling"));
215               var dif = Math.abs(now - data.lastModified);
216               if(dif > timeout) {
217                 xt.log("Removing expired session for user '%@' (%@ : %@)".f(data.userName, data.sid, data.ip));
218                 var key = '%@:%@:%@'.f(data.userName, data.ip, data.sid);
219                 xt.cache.del(key);
220               }
221               else {
222                 var timeleft = Math.round(((Math.abs(timeout-dif)/1000)/60)*100)/100;
223                 xt.log("%@ : %@ => %@ minutes left".f(data.userName, data.sid, timeleft));
224               }
225             });
226           }
227         });
228       }
229     });
230   },
231 
232   /** @private */
233   init: function() {
234     
235     // set an interval for polling the cache for these expired sessions
236     this.__pollingTimer__ = setInterval(this.pollCache, 60000 /* 1 minute */);
237   },
238 
239   /** @private */
240   className: 'xt.sessionStore'
241 
242 });
243 
244 process.once('xtBootstrapped', function() {
245   
246   // grab the session-timeout from the configuration and calculate
247   // that for the appropriate number of milliseconds
248   xt.sessionStore.sessionTimeout = xt.opts.t ? xt.opts.t * 60000 : 900000; // default to 15 minutes
249   
250   // share this information with the world...  
251   xt.log("Session timeout has been set to %@ minutes (%@ milliseconds)".f(xt.opts.t, xt.sessionStore.sessionTimeout));
252   
253   // go ahead and poll it from the start to clear out any invalid
254   // sessions since the node server is just starting up
255   process.once('xtCacheAvailable', xt.sessionStore.pollCache);
256 });
