  2 /** @namespace
  3   Routers provide a mechanism (and the possibility for) custom handling for
  4   specific types of requests. They are instanced and used by individual
  5   servers to handle their incoming requests. In their most basic form routers
  6   are comprised of a single-entry API (handle) that the server passes the
  7   opened server response object and the native server request object.
  8 */
  9 xt.router = xt.object.extend(
 10   /** @lends xt.router.prototype */ {
 11 
 12   /**
 13     This method is used by the router to determine if it knows of any
 14     routes capable of handling the request. While it can be overloaded
 15     it should generally not be modified or called directly.
 16 
 17     @param {Object} xtr The xt response object.
 18 
 19     @method
 20   */
 21   route: function(xtr) {
 22     
 23     
 24     // xt.debug("in xt.router.route");
 25     
 26     
 27     var r = this.get('routes');
 28     if(xt.none(xtr))
 29       return issue(xt.warning(
 30         "Cannot route non-xt.response object or none present"));
 31 
 32     // favicon requests are dropped arbitrarily for now
 33     if(xtr.get('info.url') === 'favicon.ico')
 34       return xtr.close();
 35 
 36     if(xt.none(r) || r.keys().length <= 0)
 37       issue(xt.close(
 38         "Routes were unavailable for the request on server: " +
 39         "{server}".f({ server: this.get('server.name') || 'UNKNOWN' }), xtr));
 40     else if(r[xtr.route])
 41       r[xtr.route].handle(xtr);
 42     else {
 43       xtr.write(
 44         "I can't believe I'm even humoring you with this response, " +
 45         "you have no idea what you're asking for.");
 46       issue(xt.close(
 47         "Unable to handle the route requested, {route}, on server {server}".f(
 48           { server: this.get('server.name') || 'UNKNOWN', route: xtr.route }), xtr));
 49     }
 50   },
 51 
 52   /**
 53     This method is called by servers to handle incoming requests.
 54     It creates a xt response object that in turn emits an event on the
 55     router to execute the route when it is ready.
 56 
 57     @param {Object} res The server response object.
 58     @param {Object} req The server request object.
 59 
 60     @method
 61   */
 62   handle: function(req, res) {
 63     
 64    // xt.debug("in xt.router.handle");
 65     
 66     var xtr = xt.response.create({
 67       response: res,
 68       request: req,
 69       router: this,
 70       server: this.get('server')
 71     });
 72   },
 73 
 74   /**
 75     Routes is an array of xt route objects.
 76   */
 77   routes: [],
 78 
 79   /**
 80     A router finds the routes it represents and creates a has it uses
 81     to lookup paths to the appropriate xt route object upon handling
 82     a client request.
 83 
 84     @private
 85   */
 86   init: function() {
 87     var r = this.get('routes'),
 88         n = this.routes = {}; // note the replacement!
 89     if(xt.none(r) || r.length <= 0) return;
 90     _.each(r, function(o, k) {
 91 
 92       // grab the routes specified on the xt route object
 93       // and then map each of those back to the route
 94       // so when they are encountered they can be explicitly
 95       // accessed
 96       var p = o.get('routes');
 97       if(xt.none(p)) return;
 98       _.each(p, function(u, k) {
 99         n[u] = o; // assign unique entries by path => route object
100       }, this);
101     }, this);
102   },
103 
104   /** @private */
105   className: 'xt.router'
106 });
107 
108 xt.router_default = xt.router.create();
109 
110 xt.mixin(xt.router,
111   /** @lends xt.router */ {
112 
113   loadRouters: function() {
114 
115     xt.log("Loading available routers (from {dir})".f({ dir: 
116       xt.fs.shorten(xt.fs.normalize(__dirname + '/../routers'), 5) }));
117 
118     // chdir to modules directory for convenient
119     // relative pathing
120     process.chdir(__dirname);
121 
122     // find all of the available files in the routes
123     // directory
124     xt.fs.directoryFiles('../routers', function(files) {
125 
126       var num = files.length;
127 
128       // this might should throw a fatal as there really isn't much that
129       // can be done when no routes are available
130       if(num <= 0)
131         return issue(xt.warning("There are no routers available to the server."));
132 
133       // iterate over the files and require them so they can be executed
134       _.each(files, function(v, k) {
135         require('../routers/%@'.f(v));
136         --num; if(num == 0) {
137           process.emit('xtRoutersLoaded');
138           process.emit('xtServerReady');
139         }
140       }, this);
141 
142     }, 'js');
143 
144   }
145 
146 });
147 
148 process.once('xtRoutesLoaded', xt.router.loadRouters);
