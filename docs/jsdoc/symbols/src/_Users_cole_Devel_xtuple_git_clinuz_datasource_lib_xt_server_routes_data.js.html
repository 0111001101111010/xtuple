  2 /** @class
  3   All client data requests are directed through this route.
  4 */
  5 xt.dataRoute = xt.route.create(
  6   /** @lends xt.dataRoute.prototype */ {
  7 
  8   /**
  9     All client data requests will be passed with a payload that
 10     indicates to the datasource what handler to use based on
 11     the request type.
 12 
 13     @param {Object} xtr The xt response object.
 14     @method
 15   */
 16   handle: function(xtr) {
 17     
 18     // xt.debug("in xt.dataRoute.handle");
 19     
 20     var t = xtr.get('info.json.requestType');
 21 
 22     if(xt.none(t))
 23       return xtr.write(
 24         "I can't believe I'm even humoring you with this response, ",
 25         "you have no idea what you're asking for.").close();
 26 
 27     // xt.debug("The request was type: %@".f(t)); 
 28 
 29     // attempt to find a functor to handle the request type
 30     var f = this.find(t);
 31 
 32     if(xt.none(f))
 33       return issue(xt.close("Could not handle request type, %@".f(t), xtr));
 34 
 35 
 36     // if the functor requires a session, it must be loaded before
 37     // it is allowed to handle the request
 38     if(f.target.get('needSession')) {
 39       
 40       // xt.debug("needSession was true for %@".f(f.target.toString()));
 41       
 42       xtr .set('useSession', YES)
 43           .set('handler', f.target);
 44 
 45     }
 46       
 47     // this was stored as hash { handles: '{type}', target: {object} }
 48     else {
 49       // xt.debug("DID NOT NEED SESSION?!?!");
 50       f.target.handle(xtr);
 51     }
 52   },
 53 
 54   /**
 55     Finds the functor that handles the request type requested and
 56     returns it or null if none is found.
 57 
 58     @param {String} req The request type.
 59     @returns {Object} The functor that handles the request type or null.
 60   */
 61   find: function(req) {
 62     var s = this.get('functors'), f;
 63     f = _.filter(s, function(v) {
 64       return v.handles === req;
 65     }, this);
 66     return f[0] ? f[0] : null;
 67   },
 68 
 69   /**
 70     Just one route, data.
 71     @property
 72   */
 73   routes: [ 'data' ],
 74 
 75   /**
 76     Here we've overloaded init so that we can grab any available
 77     handlers for the record types without explicitly knowing
 78     about them.
 79 
 80     @private
 81   */
 82   init: function() {
 83     var s = this;
 84     process.chdir(__dirname);
 85     xt.fs.directoryFiles('../functors', function(files) {
 86       var num = files.length;
 87       if(num <= 0)
 88         return issue(
 89           xt.warning("There were no functors to load by the data route handler."));
 90       _.each(files, function(v, k) {
 91         require('../functors/%@'.f(v));
 92         --num; if(num == 0) this.emit('xtFunctorsLoaded');
 93       }, s); // note the context
 94     }, 'js');
 95   },
 96 
 97   /**
 98     Effectively just checks to make sure we've loaded some functor's
 99     otherwise we're in trouble.
100 
101     @private
102   */
103   _finalize_init: function() {
104     this.set('functors', xt.functors);
105     xt.log("%@ functors were found and loaded for data route".f(xt.functors.length)); 
106   }.by('xtFunctorsLoaded')
107 
108 });
