  2 /** @namespace
  3   This is the primary namespace for the database driver and
  4   functionality of the framework.
  5 */
  6 
  7 xt.database = xt.object.create(
  8   /** @lends xt.database.prototype */ {
  9 
 10   /**
 11     Takes parameter(s) to construct a PostgreSQL connection
 12     string. Can be passed an ordered number of parameters
 13     or a hash of key/value entries for the string with the
 14     parts user, password, host, port and database (all required).
 15     Note, the string contains the password for the given
 16     database user!
 17 
 18     @param {Object|String} user A hash with all parameters or the username.
 19     @param {String} [password] The database password for the user.
 20     @param {String} [host] The hostname for the database.
 21     @param {Number} [port] The port for the database.
 22     @param {String} [database] The name of the database to use once connected.
 23     @returns {String} The formatted connection string.
 24     @method
 25   */
 26   conString: function() {
 27     var a = args(),
 28         s = "tcp://{user}:{password}@{host}:{port}/{database}",
 29         u, p, P, h, d; 
 30     if(a[0] && xt.typeOf(a[0]) === xt.t_hash) {
 31       var a = a[0];
 32       u = a.user;
 33       p = a.password;
 34       P = a.port;
 35       h = a.host;
 36       d = a.database;
 37     }
 38     else {
 39       u = a[0];
 40       p = a[1];
 41       h = a[2];
 42       P = a[3];
 43       d = a[4];
 44     }
 45     return s.f({
 46       user: u,
 47       password: p,
 48       host: h,
 49       port: P,
 50       database: d
 51     });
 52   },
 53 
 54   /**
 55     Returns the default connection string for the database.
 56     Typically not useful outside the context of the database
 57     controller and should not be accessed globally.
 58 
 59     @property
 60   */
 61   defaultString: function() {
 62     var s = this._defaultString;
 63     if(xt.none(s))
 64       s = this._defaultString = this.conString(xt.opts);
 65     return s;
 66   }.property(),
 67 
 68   /**
 69     Queries the database. Expects a ready SQL query string and
 70     a valid callback to handle errors and responses. It is important
 71     to note that this is not a final state for this method and
 72     the API is volatile with a strong likelyhood that it will change
 73     dramatically. This method is not intended to be executed
 74     arbitrarily but is instead used by xt session objects to
 75     execute specific types of queries.
 76 
 77     @param {String} query The SQL query string ready to execute.
 78     @param {Function} callback The callback that can also handle errors.
 79       If the callback was passed in from a xt session object then it
 80       will be capable of calling `this` from the context of the session.
 81     @method
 82   */
 83   query: function() {
 84     var a = arguments;
 85     if(
 86       a.length != 2
 87       || xt.typeOf(a[0]) !== xt.t_string
 88       || xt.typeOf(a[1]) !== xt.t_function
 89     ) return;
 90     var q = a[0],
 91         f = a[1],
 92         s = a[2] ? this.conString(a[2]) : this.get('defaultString');
 93     xt.pg.connect(s, function(e, c) {
 94       if(e) throw xt.fatal("Failed to connect to database", e);
 95       c.query(q, f);
 96     });
 97   }
 98 
 99 });
