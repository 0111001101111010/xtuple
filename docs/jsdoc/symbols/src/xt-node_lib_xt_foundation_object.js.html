  5   The xt.object class is the base class for nearly all custom
  6   objects in the xt framework for node.js server.

  8   @param {Object} [hashes] Hashes or a hash to extend from.
  9   @returns {Object} A new instance of an xt.object or child of xt.object.

 11 xt.object = function() { 
 12   this.__super__ = xt.object.prototype;
 13   return xt.__init__.apply(this, arguments[0]);

 18 //
 19 // These methods make using xt.object's more convenient and allow

 21 xt.mixin(xt.object,
 22   /** @lends xt.object */ {
 23   
 24   /**
 25     Creates a new object based on this object type. Accepts hashes
 26     for extension that will create either a new class type as a singleton
 27     or simply a new object of this type with properties initialized to
 28     the given values.
 29     
 30     @param {Object} [hashes] Mixins to be added to the object.
 31     @returns {Object} An instance of the [extended] object.
 32   */
 33   create: function create() { 
 34     var c = this, r = new c(arguments); return r; 
 35   },
 36   
 37   /**
 38     Extends the base object with the given properties. It creates a new class
 39     that inherits from the base. Can take any number of key-value hashes to
 40     extend the base with that redefine it as a new class or simply the same base
 41     class with different properties at initialization.
 42     
 43     @param {Object} [hashes] Mixins to be added to the object.
 44     @returns {Object} An extended prototype.
 45   */
 46   extend: function extend() {
 47     var r = function() { return xt.__init__.apply(this, arguments[0]); },
 48         l = arguments.length,
 49         i = 0, k, p;
 50     for(k in this)
 51       if(!this.hasOwnProperty(k)) continue;
 52       else r[k] = this[k];
 53     if(this.hasOwnProperty('toString')) r.toString = this.toString;
 54     p = (r.prototype = xt.__sire__(this.prototype));
 55     for(; i<l; ++i)
 56       xt.__proto_extend__(p, arguments[i], this.prototype);
 57     p.constructor = r;
 58     p.__super__ = this.prototype;
 59     return r;
 60   },
 61   
 62   /**
 63     Simple mechanism to differentiate between instance and class.
 64     
 65     @return {String} Formatted name of the class.
 66   */
 67   toString: function() { 
 68     var c = this.className
 69       || this.prototype.className;
 70     return "[ CLASS: %@ ]".format(c);
 71   }

 84   /**
 85     Set a property on the given object to be equal to a specified
 86     value. This is essentially meta-data of the object and can be
 87     referenced and manipulated many ways but also allows for types
 88     of notifications to take place.
 89     
 90     @param {String} key The key/path for the property to set (relative).
 91     @param {Object} value The object (any type) to place at path.
 92     @returns {Object} Callee for chaining.
 93   */
 94   set: function() {
 95     return xt.__set__.apply(this, arguments);
 96   },
 97   
 98   /**
 99     Retrieve a value for the given property path on the object.
100     
101     @param {String} key The key to lookup and retrieve.
102     @returns {Object|Boolean} The value for the key or undefined if not found.
103   */
104   get: function() {
105     return xt.__get__.apply(this, arguments);
106   },
107   
108   /**
109     Each class should have a name. This name is used in several
110     ways of matching types, retrieving classes from strings,
111     comparisons, etc.
112 
113     @private
114   */
115   className: 'xt.object',
116   
117   /**
118     The map of custom event handlers and their methods to execute
119     for removal purposes.
120 
121     @private
122   */ 
123   _events: null,
124 
125   /**
126     Adds an event handler to process when a particular event is
127     encountered.
128 
129     Note that it currently ONLY SUPPORTS ONE METHOD FOR ANY GIVEN EVENT!
130 
131     @param {String} event The event to listen for.
132     @param {Function} func The function to execute when the event is encountered.
133     @returns {Object} Receiver
134   */
135   addEvent: function(e, f) {
136     if(!this._events) this._events = {};
137 
138     // keep track of the event and what method it is mapped to
139     this._events[e] = f;
140 
141     // this is a hack because there wasn't enough time to track down
142     // why multiple entries are occurring (obviously possible but
143     // in the test cases it was even though it should not have been)
144     this.removeAllListeners(e);
145 
146     // should now be solo entry for event although this is less than
147     // ideal since there should be able to be more than 1 observer
148     // for any single event...for now this is an arbitrary limitation
149     // that needs to be revisited
150     this.on(e, f);
151   },
152 
153   /**
154     Allows an event to be removed.
155 
156     @param {String} event The name of the event to stop listening for.
157     @returns {Object} Receiver
158   */
159   removeEvent: function(e) {
160     if(this._events[e]) {
161       this.removeListener(e, this._events[e]);
162       delete this._events[e];
163     }
164   },
165 
166   /**
167     While this method can be overridden, most cases it won't be.
168     It provides a consistent, accurate and relatively convenient
169     human-readable output for debugging (mostly) but sometimes
170     comparisons.
171     
172     @returns {String} The string for the object.
173     @private
174   */
175   toString: function toString() {
176     var c = this.className || 'NOCLASS',
177         g = this.uid || 'NOUID';
178     return "[ %@ (%@) ]".format(c, g);
179   },

181   /**
182     Length is an arbitrary (and somewhat ambiguous) property
183     that should really only be available on enumerable
184     implementations but, for now, is available on all xt objects
185     while not always useful or meaningful.
186 
187     It attempts to determine a meaningful value based solely on
188     properties owned by the object instance and should be accessed
189     via the get method.
190 
191     @example 
192       var len = this.get('length')
193 
194     @returns {Number} The number of unique properties owned by the object.
195     @public
196   */
197   length: function() {
198     var c = [];
199     for(var k in this)
200       if(!this.hasOwnProperty(k)) continue;
201       else c.push(k);
202     return c.length;
203   }.property(),
204 }
