  3 /** @class
  4 */
  5 xt.requestSession = xt.functor.create(
  6   /** @lends xt.requestSession.prototype */ {
  7 
  8   /** @method */
  9   handle: function(xtr) {
 10     var payload = xtr.get('payloadJSON'),
 11         u = payload.username,
 12         p = payload.password,
 13         s = payload.sid;
 14     if(xt.none(p)) return issue(xt.close("Must have password", xtr)); // need real errors
 15     if(xt.none(u)) return issue(xt.close("Must have username", xtr)); // need real errors
 16     if(s) xt.sessionStore.validateSession(xtr, this.success);
 17     else xt.sessionStore.createSession(xtr, this.success); 
 18   },
 19 
 20   success: function(session, xtr) {
 21     if(xt.none(session))
 22       return issue(xt.close("Reported success on load/creation of session but no session available", xtr));
 23     
 24     xtr.write(session.get('data')).close();
 25   },
 26 
 27   /** @property */
 28   handles: 'requestSession',
 29   
 30   needSession: NO,
 31 
 32   /** @private */
 33   className: 'xt.requestSession'
 34 
 35 });
