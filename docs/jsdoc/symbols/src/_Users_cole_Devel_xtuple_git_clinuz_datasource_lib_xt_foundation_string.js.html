  2 /** @namespace
  3   This is a collection of methods that are available for
  4   string manipulation. Some are optimizations and some are merely
  5   for convenience. Some are available on the native String object
  6   prototype and are marked private here. Any public method can
  7   be used from the namespace. The methods added to the prototype
  8   call these methods so as not to add so much additional code to
  9   each individual string when they can share.
 10 */
 11 
 12 xt.string = { /** @lends xt.string.prototype */
 13 
 14   /**
 15     Has the ability to replace keys in the string with
 16     the contents provided to it. It can handle a hashes
 17     of key-value pairs and it will replace all keys with
 18     the matching values throughout the string (not just
 19     the first reference). It can also handle an indefinite
 20     number of strings and they will sequentially be inserted
 21     into the string replacing any `%@` found. These can be
 22     any number of arguments and be mixed and matched if need
 23     be.
 24     
 25     The keys in the string should be wrapped in curly-braces
 26     because this is how the regex will look for them.
 27     
 28     @example
 29       var s = "hello {name} you are {adjective}";
 30       s = s.format({name: 'pooBear', adjective: 'honey-lover'});
 31     
 32       or
 33     
 34       var s = "hello {name} you are %@";
 35       s = s.format({name: 'pooBear'}, 'honey-lover');
 36     
 37       both will reconcile to
 38     
 39       s => "hello pooBear you are honey-lover"
 40 
 41     @param {Object} param The key/value pairs to insert into string.
 42     @returns {String} The parsed string.
 43   */
 44   format: function format() {
 45     if(arguments.length === 0) return this;
 46     var args = Array.prototype.slice.call(arguments),
 47         l = args.length,
 48         i = 0, 
 49         self = this, t;
 50     for(; i<l; ++i) {
 51       if(xt.none(args[i])) continue;
 52       t = xt.typeOf(args[i]);
 53       if(t === xt.t_hash)
 54         self = xt.string.replaceKeys(self, args[i]);
 55       else if(t === xt.t_string || t === xt.t_number)
 56         self = self.replace(/\%@/, args[i]);
 57       else continue;
 58     }
 59     return self;
 60   },
 61   
 62   /**
 63     Capitalizes the first letter of the string and forces
 64     the remaining characters of the string to lowercase.
 65     
 66     @todo This may not work on translated strings?
 67     
 68     @returns {String} The capitalized string.
 69   */
 70   capitalize: function capitalize() {
 71     return this.charAt(0).toUpperCase()+this.slice(1).toLowerCase();
 72   },
 73   
 74   /**
 75     Trims away whitespace at the beginning and end of the string.
 76     The implementation selected (since there are a wide variety) tested to
 77     be the fastest method for normal scenarios.
 78     
 79     @returns {String} trimmed string
 80   */
 81   trim: function trim() {
 82     return this.replace(/^\s\s*/, '').replace(/\s\s*$/, '');
 83   },
 84 
 85   /**
 86     Prepends a string to the front of the current string. This
 87     is more convenient, can save a few characters, and arguably
 88     looks better in code.
 89     
 90     @returns {String} string with prepended value as prefix
 91   */
 92   pre: function pre(p) {
 93     if(xt.none(p) || xt.typeOf(p) !== xt.t_string) return this;
 94     return p + this;
 95   },
 96   
 97   /**
 98     Takes an object-hash of key-value pairs and globally replaces
 99     each key with its assigned value anywhere in the string.
100     
101     @param {String} the string to search and replace in
102     @param {Object} the hash of key-value pairs
103     @returns {String} the modified string
104   */
105   replaceKeys: function replaceKeys(s, o) {
106     if(xt.none(o) || xt.typeOf(o) !== xt.t_hash) return s;
107     if(xt.typeOf(s) !== xt.t_string) return '';
108     var r, k, m, v;
109     for(k in o) {
110       m = '{%@}'.format(k);
111       r = new RegExp(m, 'g');
112       v = o[k];
113       s = s.replace(r, v)
114     }
115     return s;
116   },
117 
118  /** 
119     Camelizes current string, removing '_' and ' '.
120 
121     @returns {String} the modified (camelized) string
122  */
123   camelize: function camelize() {
124     var self = this, ret, first, lower; 
125     ret = self.replace( (/([\s|\-|\_|\n])([^\s|\-|\_|\n]?)/g), function(self, separater, character) {
126           return character ? character.toUpperCase() : '';
127           });
128     first = ret.charAt(0);
129     lower = first.toLowerCase();
130     return first !== lower ? lower + ret.slice(1) : ret;
131   },
132 
133   /* Decamelizes current string adding '_' between (formerly) camelized values.
134 
135      @returns {String} the modified (decamelized) string
136   */
137   decamelize: function camelize() {
138     return this.replace((/([a-z])([A-Z])/g), '$1_$2').toLowerCase(); 
139   }
