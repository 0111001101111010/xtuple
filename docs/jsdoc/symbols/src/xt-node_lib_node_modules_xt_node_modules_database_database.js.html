  7 xt.database = xt.object.create(
  8   /** @scope xt.database.prototype */ {
  9 
 10   /**
 11     ...
 12   */
 13   conString: function() {
 14     var a = args(),
 15         s = 'tcp://{user}:{password}@{host}:{port}/{database}',
 16         u, p, P, h, d; 
 17     if(a[0] && xt.typeOf(a[0]) === xt.t_hash) {
 18       var a = a[0];
 19       u = a.user;
 20       p = a.password;
 21       P = a.port;
 22       h = a.host;
 23       d = a.database;
 24     }
 25     else {
 26       u = a[0];
 27       p = a[1];
 28       h = a[2];
 29       P = a[3];
 30       d = a[4];
 31     }
 32     return s.f({
 33       user: u,
 34       password: p,
 35       host: h,
 36       port: P,
 37       database: d
 38     });
 39   },
 40 
 41   defaultString: function() {
 42     var s = this._defaultString;
 43     if(xt.none(s))
 44       s = this._defaultString = this.conString(xt.opts);
 45     return s;
 46   }.property(),
 47 
 48   /**
 49     ...
 50   */
 51   query: function() {
 52     var a = arguments;
 53     if(
 54       a.length != 2
 55       || xt.typeOf(a[0]) !== xt.t_string
 56       || xt.typeOf(a[1]) !== xt.t_function
 57     ) return;
 58     var q = a[0],
 59         f = a[1],
 60         s = a[2] ? this.conString(a[2]) : this.get("defaultString");
 61     xt.pg.connect(s, function(e, c) {
 62       if(e) throw xt.fatal("Failed to connect to database", e);
 63       c.query(q, f);
 64     });
 65   }
 66 
 67 });
